Select Function

Implementation 
        function select(data, callback) {
          let local = [];
          for(let i = 0; i < data.length; ++i) {
            let element = data[i];
            if (callback(element) === true) {
              local.push(element);
            }
          }
          return local;
        }
Usage 
          var ages = [16,21,30,45,60,75];
          var young = select(ages, function(item) {
            return (item <= 25);
          });

Find Function

Implementation 
            function find(data, callback) {
              for(let i = 0; i < data.length; ++i) {
                let element = data[i];
                if (callback(element) === true) {
                  return element;
                }
              }
              return undefined;
            }
Usage 
            var nums = [1,89,45,26,57,34];
            var firstEven = find(nums, function(item) {
              return (item % 2 == 0);
            });
